Connection conexao = null;
PreparedStatement pst = null;
ResultSet rs = null;

Criação do método:

public void logar(){
    String sql = "select * from "nome da tabela no mySQL" where login = ? and senha = ?";
    try{
        pst = conexao.prepareStatement(sql);
        pst.setString(1, campoUsuario.getText());
        pst.setString(2, campoSenha.getText());
        rs = pst.executeQuery();
        
        if(rs.next()){
            MainView "NOME VARIAVEL"= new MainView();
            "VARIAVEL ACIMA".setVisible(true);
            this.dispose(); //Faz fechar a tela logo apois abrir a outra
        } else {
            JOptionPane.showMessageDialog(null, "Usuário e/ou senha inválido(s)");
        }
    }
    catch (Exception e) {
        JOptionPane.showMessageDialog(null,e);
    }
}
